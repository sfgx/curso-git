% Instalación Visual Code
\begin{tarea}{Instalación Visual Studio Code}{}
\begin{chklst}
    \item Ir a la página web:  https://code.visualstudio.com/Download
    \item Descargar la versión System Installer correspondiente a su sistema
            operativo. La versión vigente es la $1.106$
    \item Ejecutar el archivo descargado; seguir las indicacciones. Aceptar 
            las opciones por defecto.
\end{chklst}

\tcblower
\vspace{2cm}
\end{tarea}

% Instalación Git
\begin{tarea}[colframe=ggreen]{Instalación de Git}{}
\begin{chklst}
    \item Ir a la página web:  https://git-scm.com/
    \item Descargar la versión de git correspondiente a su sistema
            operativo. La versión vigente es la $2.52.0$
    \item Ejecutar el archivo descargado; seguir las indicacciones. Aceptar 
            las opciones por defecto.
\end{chklst}

\tcblower
\vspace{2cm}
\end{tarea}
